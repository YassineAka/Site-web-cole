 * ============================================
 */
 
/* Suppression des tables si BD existe deja */


DROP TABLE IF EXISTS Teacher;

/* Creation des tables */



CREATE TABLE Teacher (
    id      VARCHAR(3) NOT NULL,
    name    VARCHAR(20) NOT NULL,
	firstName    VARCHAR(20) NOT NULL,

     CONSTRAINT TeacherPK PRIMARY KEY (id)
);
